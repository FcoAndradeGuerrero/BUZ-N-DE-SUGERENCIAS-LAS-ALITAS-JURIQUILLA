// ── Tab switching ──

function switchTab(tab) {

  document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));

  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));

  document.getElementById('view-' + tab).classList.add('active');

  event.target.classList.add('active');

  if (tab === 'responses') renderResponses();

}


// ── Anon toggle ──

document.getElementById('anon').addEventListener('change', function() {

  document.getElementById('name-field').style.display = this.checked ? 'none' : 'block';

});


// ── Star rating ──

const starLabels = document.querySelectorAll('.star-label');

starLabels.forEach((label, i) => {

  label.addEventListener('mouseover', () => {

    starLabels.forEach((l, j) => l.classList.toggle('active', j <= i));

  });

  label.addEventListener('mouseout', () => {

    const checked = document.querySelector('input[name="stars"]:checked');

    const val = checked ? parseInt(checked.value) - 1 : -1;

    starLabels.forEach((l, j) => l.classList.toggle('active', j <= val));

  });

});


// ── Submit ──

function submitForm() {

  const msg = document.getElementById('mensaje').value.trim();

  if (!msg) {

    document.getElementById('mensaje').focus();

    document.getElementById('mensaje').style.borderColor = 'var(--accent)';

    setTimeout(() => document.getElementById('mensaje').style.borderColor = '', 1500);

    return;

  }


  const tipo = document.querySelector('input[name="tipo"]:checked')?.value || 'otro';

  const area = document.getElementById('area').value;

  const stars = document.querySelector('input[name="stars"]:checked')?.value || null;

  const anon = document.getElementById('anon').checked;

  const nombre = anon ? null : document.getElementById('nombre').value.trim();


  const entry = { tipo, area, mensaje: msg, stars, anon, nombre, fecha: new Date().toISOString() };

  const all = JSON.parse(localStorage.getItem('buzon_responses') || '[]');

  all.unshift(entry);

  localStorage.setItem('buzon_responses', JSON.stringify(all));


  document.getElementById('form-wrapper').style.display = 'none';

  document.getElementById('success').style.display = 'block';

}


function resetForm() {

  document.getElementById('mensaje').value = '';

  document.getElementById('area').value = '';

  document.querySelector('input[name="tipo"][value="sugerencia"]').checked = true;

  document.querySelectorAll('input[name="stars"]').forEach(r => r.checked = false);

  starLabels.forEach(l => l.classList.remove('active'));

  document.getElementById('anon').checked = true;

  document.getElementById('name-field').style.display = 'none';

  document.getElementById('nombre').value = '';

  document.getElementById('form-wrapper').style.display = 'block';

  document.getElementById('success').style.display = 'none';

}


// ── QR ──

let qrInstance = null;

function generateQR() {

  const url = document.getElementById('url-input').value.trim();

  if (!url) { alert('Por favor ingresa una URL válida.'); return; }


  const container = document.getElementById('qr-container');

  container.innerHTML = '';

  if (qrInstance) { qrInstance.clear(); }


  qrInstance = new QRCode(container, {

    text: url,

    width: 200,

    height: 200,

    colorDark: '#1a1a2e',

    colorLight: '#ffffff',

    correctLevel: QRCode.CorrectLevel.H

  });


  document.getElementById('qr-actions').style.display = 'flex';

}


function downloadQR() {

  const canvas = document.querySelector('#qr-container canvas');

  if (!canvas) { alert('Primero genera el QR.'); return; }

  const link = document.createElement('a');

  link.download = 'qr-buzon-sugerencias.png';

  link.href = canvas.toDataURL();

  link.click();

}


// ── Responses ──

function renderResponses() {

  const all = JSON.parse(localStorage.getItem('buzon_responses') || '[]');

  const list = document.getElementById('resp-list');

  if (!all.length) {

    list.innerHTML = '<div class="resp-empty">📭 Aún no hay respuestas</div>';

    return;

  }

  const icons = { sugerencia:'💡', queja:'😤', felicitacion:'🌟', otro:'💬' };

  const labels = { sugerencia:'Sugerencia', queja:'Queja', felicitacion:'Felicitación', otro:'Otro' };

  list.innerHTML = all.map(r => {

    const d = new Date(r.fecha);

    const fecha = d.toLocaleDateString('es-MX', { day:'2-digit', month:'short', year:'numeric', hour:'2-digit', minute:'2-digit' });

    const stars = r.stars ? '★'.repeat(parseInt(r.stars)) + '☆'.repeat(5-parseInt(r.stars)) : '';

    return `

      <div class="resp-card">

        <div class="resp-top">

          <span class="resp-type type-${r.tipo}">${icons[r.tipo]} ${labels[r.tipo]}</span>

          <span class="resp-date">${fecha}</span>

        </div>

        ${r.area ? `<div class="resp-area">📍 ${r.area}</div>` : ''}

        ${stars ? `<div class="resp-stars">${stars}</div>` : ''}

        <div class="resp-text">${r.mensaje}</div>

        ${r.anon ? '<div class="resp-anon">Enviado anónimamente</div>' : ''}

        ${r.nombre ? `<div class="resp-name">👤 ${r.nombre}</div>` : ''}

      </div>`;

  }).join('');

}


function clearResponses() {

  if (confirm('¿Borrar todas las respuestas? Esta acción no se puede deshacer.')) {

    localStorage.removeItem('buzon_responses');

    renderResponses();

  }

}
